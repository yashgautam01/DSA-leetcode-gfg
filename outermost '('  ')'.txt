
class Solution {
    public static String removeOuter(String s) {
    
    
        // code here
        
        StringBuilder sb  = new StringBuilder();
        int j=0;
        char[] ch = s.toCharArray();
        for(int i=0;i<ch.length;i++){
            if(ch[i]==')')j--;
            if(j!=0 ) sb.append(ch[i]);
            
            
            if(ch[i]=='(')j++;
            
        }
        return sb.toString();
        
    }
}
