class Solution {
    public int findMaxConsecutiveOnes(int[] a) {
        int max1=-1;
        int n = a.length;
        int cur = 0;
        for(int i =0;i<n;i++){
            if(a[i]==1 ){
                cur++;
            }else{
            
            cur=0;
            }
            max1 = Math.max(max1,cur); 
        }
        return max1;
    }
}
