class Solution {

    public int[] twoSum(int[] nums, int target) {
       int n = nums.length;
       
       
       int res[]=new int[2];
       Map<Integer,Integer> h = new HashMap<>();
       for(int i=0;i<n;i++){

       
       
           if(h.containsKey(target-nums[i])){
           
               res[0]=h.get(target-nums[i]);
               res[1] = i;
               return res;
           }
           h.put(nums[i],i);
       }
       return res;

    }
}
